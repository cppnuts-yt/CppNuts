	#include<iostream>
	#include<string>
	#include<sstream>
	using namespace std;
	template<typename T>
	string itos(T i)	// convert int to string
	{
		stringstream s;
		s << i;
		return s.str();
	}

	int main()
	{
		float i = 127.4;
		string ss = itos(i);
		const char* p = ss.c_str();

		cout << ss << " " << p << "\n";
	}
